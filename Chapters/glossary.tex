\newglossaryentry{Alt-Ergo}{
  name={Alt-Ergo},
  %sort={Alt-Ergo},
  description={Open-source and SMT-based automatic solver of mathematical formulas, capeable of reasoning about records, 
  enumerates, polymorphism, linear arithmetic and others theories.}
}
\newglossaryentry{Cameleer}{
  name={Cameleer},
  %sort={Cameleer},
  description={An automated deductive verification tool for OCaml programs using GOSPEL annotations.}
}
\newglossaryentry{CVC5}{
  name={CVC5},
  %sort={CVC5},
  description={SMT solver that supports various theories, including separation logic, bit-vectors, set theory, quantifiers, 
  among others.}
}
\newglossaryentry{GOSPEL}{
  name={GOSPEL},
  %sort={GOSPEL},
  description={A specification language for the OCaml language, intended to be used in various purposes. The acronym stands 
  for Generic OCaml SPEcification Language.}
}
\newglossaryentry{OCaml}{
  name={OCaml},
  %sort={OCaml},
  description={A Pragmatic functional programming language with roots in academia and growing commercial use. It supports
  highly complex features, such as generational garbage collection, type inference, parametric polymorphism, an efficient
  compiler, among many others.}
}
\newglossaryentry{Why3}{
  name={Why3},
  %sort={Why3},
  description={A platform for deductive program verification that relies on external provers.}
}
\newglossaryentry{WhyML}{
  name={WhyML},
  %sort={WhyML},
  description={The programming and specification language used in the Why3 platform. It contains many features commonly 
  present in modern functional languages, and supports built-in annotations to verification purposes.}
}
\newglossaryentry{CakeML}{
  name={CakeML},
  %sort={CakeML},
  description={A functional programming language based on ML, designed with a formally verified compiler. It aims to 
  provide a trustworthy foundation for building secure and reliable software, particularly in verified systems.}
}
\newglossaryentry{CompCert}{
  name={CompCert},
  %sort={CakeML},
  description={A realistic and verified compiler, which enables formal reasoning and machine-checked proofs of correctness 
  for each compilation phase.}
}
\newglossaryentry{Z3}{
  name={Z3},
  %sort={Z3},
  description={SMT solver from Microsoft Research. It combines multiple theories, such as, arrays, arithmetic, bit-vectors.
  }
}

\newglossaryentry{Gantt Chart}{
  name={Gantt Chart},
  %sort={Gantt Chart},
  description={A bar chart used in project management for scheduling tasks.}
}
